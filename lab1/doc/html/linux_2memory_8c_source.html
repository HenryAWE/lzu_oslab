39 /*
40  * Well, here is one of the most complicated functions in mm. It
41  * copies a range of linerar addresses by copying only the pages.
42  * Let's hope this is bug-free, 'cause this one I don't want to debug :-)
43  *
44  * Note! We don't copy just any chunks of memory - addresses have to
45  * be divisible by 4Mb (one page-directory entry), as this makes the
46  * function easier. It's used only by fork anyway.
47  *
48  * NOTE 2!! When from==0 we are copying kernel space for the first
49  * fork(). Then we DONT want to copy a full page-directory entry, as
50  * that would lead to some serious memory waste - we just copy the
51  * first 160 pages - 640kB. Even that is more than we need, but it
52  * doesn't take any more memory - we don't copy-on-write in the low
53  * 1 Mb-range, so the pages can be shared with the kernel. Thus the
54  * special case for nr=xxxx.
55  */
56 int copy_page_tables(unsigned long from,unsigned long to,long size)
57 {
58  unsigned long * from_page_table;
59  unsigned long * to_page_table;
60  unsigned long this_page;
61  unsigned long * from_dir, * to_dir;
62  unsigned long new_page;
63  unsigned long nr;
64 
65  if ((from&0x3fffff) || (to&0x3fffff))
66  panic("copy_page_tables called with wrong alignment");
67  from_dir = (unsigned long *) ((from>>20) & 0xffc); /* _pg_dir = 0 */
68  to_dir = (unsigned long *) ((to>>20) & 0xffc);
69  size = ((unsigned) (size+0x3fffff)) >> 22;
70  for( ; size-->0 ; from_dir++,to_dir++) {
71  if (1 & *to_dir)
72  panic("copy_page_tables: already exist");
73  if (!(1 & *from_dir))
74  continue;
75  from_page_table = (unsigned long *) (0xfffff000 & *from_dir);
76  if (!(to_page_table = (unsigned long *) get_free_page()))
77  return -1; /* Out of memory, see freeing */
78  *to_dir = ((unsigned long) to_page_table) | 7;
79  nr = (from==0)?0xA0:1024;
80  for ( ; nr-- > 0 ; from_page_table++,to_page_table++) {
81  this_page = *from_page_table;
82  if (!this_page)
83  continue;
84  if (!(1 & this_page)) {
85  if (!(new_page = get_free_page()))
86  return -1;
87  read_swap_page(this_page>>1, (char *) new_page);
88  *to_page_table = this_page;
89  *from_page_table = new_page | (PAGE_DIRTY | 7);
90  continue;
91  }
92  this_page &= ~2;
93  *to_page_table = this_page;
94  if (this_page > LOW_MEM) {
95  *from_page_table = this_page;
96  this_page -= LOW_MEM;
97  this_page >>= 12;
98  mem_map[this_page]++;
99  }
100  }
101  }
102  invalidate();
103  return 0;
104 }

235 /*
236  * try_to_share() checks the page at address "address" in the task "p",
237  * to see if it exists, and if it is clean. If so, share it with the current
238  * task.
239  *
240  * NOTE! This assumes we have checked that p != current, and that they
241  * share the same executable or library.
242  */
243 static int try_to_share(unsigned long address, struct task_struct * p)
244 {
245  unsigned long from;
246  unsigned long to;
247  unsigned long from_page;
248  unsigned long to_page;
249  unsigned long phys_addr;
250 
251  from_page = to_page = ((address>>20) & 0xffc);
252  from_page += ((p->start_code>>20) & 0xffc);
253  to_page += ((current->start_code>>20) & 0xffc);
254 /* is there a page-directory at from? */
255  from = *(unsigned long *) from_page;
256  if (!(from & 1))
257  return 0;
258  from &= 0xfffff000;
259  from_page = from + ((address>>10) & 0xffc);
260  phys_addr = *(unsigned long *) from_page;
261 /* is the page clean and present? */
262  if ((phys_addr & 0x41) != 0x01)
263  return 0;
264  phys_addr &= 0xfffff000;
265  if (phys_addr >= HIGH_MEMORY || phys_addr < LOW_MEM)
266  return 0;
267  to = *(unsigned long *) to_page;
268  if (!(to & 1))
269  if (to = get_free_page())
270  *(unsigned long *) to_page = to | 7;
271  else
272  oom();
273  to &= 0xfffff000;
274  to_page = to + ((address>>10) & 0xffc);
275  if (1 & *(unsigned long *) to_page)
276  panic("try_to_share: to_page already exists");
277 /* share them: write-protect */
278  *(unsigned long *) from_page &= ~2;
279  *(unsigned long *) to_page = *(unsigned long *) from_page;
280  invalidate();
281  phys_addr -= LOW_MEM;
282  phys_addr >>= 12;
283  mem_map[phys_addr]++;
284  return 1;
285 }

288  * share_page() tries to find a process that could share a page with
289  * the current one. Address is the address of the wanted page relative
290  * to the current data space.
291  *
292  * We first check if it is at all feasible by checking executable->i_count.
293  * It should be >1 if there are other tasks sharing this inode.
294  */
295 static int share_page(struct m_inode * inode, unsigned long address)
296 {
297  struct task_struct ** p;
298 
299  if (inode->i_count < 2 || !inode)
300  return 0;
301  for (p = &LAST_TASK ; p > &FIRST_TASK ; --p) {
302  if (!*p)
303  continue;
304  if (current == *p)
305  continue;
306  if (address < LIBRARY_OFFSET) {
307  if (inode != (*p)->executable)
308  continue;
309  } else {
310  if (inode != (*p)->library)
311  continue;
312  }
313  if (try_to_share(address,*p))
314  return 1;
315  }
316  return 0;
